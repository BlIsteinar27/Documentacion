# 1. Crea la carpeta del proyecto
mkdir mi-proyecto-poo
cd mi-proyecto-poo

# 2. Crea las carpetas src y lib
mkdir src lib

# 3. Crea composer.json
composer init

# 4. Durante composer init, pon:
#   - name: "mi-proyecto/desafio"
#   - description: opcional
#   - type: library
#   - autoload: PSR-4 → App\ → src/
#   - Agrega manualmente después: Libreria\ → lib/

# 5. Edita composer.json y asegúrate de que autoload quede así:
"autoload": {
    "psr-4": {
        "App\\": "src/",
        "Libreria\\": "lib/"
    }
}

# 6. Instala Composer
composer install

# 7. Crea tus clases en src/ y lib/

# 8. Crea index.php y prueba
# 9. Ejecuta: php index.php
