namespace App\Models;
class User {
  public function __construct(protected ?int $id, protected string $name, protected string $email) {
    $this->setEmail($email);
  }
  public function setEmail(string $email){ if (!filter_var($email,FILTER_VALIDATE_EMAIL)) throw new \InvalidArgumentException("email"); $this->email=$email; }
  public function toArray(): array { return ['id'=>$this->id,'name'=>$this->name,'email'=>$this->email]; }
}
