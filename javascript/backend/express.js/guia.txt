1. npm init -> para iniciar proyecto de node (dar enter para todas configuraciones por defecto)
2. crear carpeta src con archivos index.js, app.js y config.js
3. pegar este contenido en config.js (donde van a estar todas la variables globales)
modulo.exports = {
    app: {
        port: proccess.env.PORT || 3000,
    }
};
4.pegar este contenido en index.js 
const app = require('./app');

app.listen(app.get('port'), () => {
    console.log(`Server is running on port ${app.get('port')}`);
});
5. ir a package.json y sustituir esto 
 "scripts": {
    "test": "echo \"Error: no test specified\" && exit 1"
  },
  
  por esto 
  "scripts": {
    "dev": "nodemon /src/index.js"
  },

9. crear src/red/respuestas.js y pegar este contenido (son helpers para respuestas )
exports.success = function (req, res, mensaje = '', status = 200) {
   
  res.status(status).send({
    error: false,
    status: status,
    body: mensaje,
  });
};

exports.error = function (req, res, mensaje ='Error interno del servidor', status = 500) {

  res.status(status).send({
    error: true,
    status: status,
    body: mensaje,
  });
};
